void main(){
  
    /*dynamic datatype is a datatype 
      where we can put any datatypes which is not needed to be specified
   */
  
    dynamic name= "Rashik";
    dynamic age="18";
    dynamic address="false";
    dynamic array=[1,2,3];
    dynamic result="2.12";
  
  print(name);
  print(age);
  print(address);
  print(array);
  print(result);
  
  
  
}
